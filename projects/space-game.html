        public void move(float dt, ControlType ct, PApplet p) {
                switch (ct) {
                    case POSITION:
                        break;

                    case VELOCITY:
                        vel.add(acc.mult(dt));
                        pos.add(PVector.mult(vel, dt));
                        // acc.mult(0);
                        break;

                    case FORCE:
                        force(dt, p); 
                        break;
                }
            }
